HELLO WORLD IN DIFFERENT LANGUAGES

Assembly x86


                    section .text
                        global _start

                    section .data
                    msg db  'Hello, world!!!', 0xa
                    len equ $ - msg

                    section .text

                    _start:

                    mov edx, len
                    mov ecx, msg
                    mov ebx, 1
                    mov eax, 4
                    int 0x80

                    mov ebx, 0
                    mov eax, 1
                    int 0x80

C++


                    #include <iostream>

                    using namespace std;

                    int main() {
                        cout << "Hello, world!!!" << endl;
                    }

PHP


                    <?php
                        echo "Hello, world!!!" . PHP_EOL;
                    ?>

JavaScript


                    console.log("Hello, world!!!")

SQL


                    SELECT "Hello, world!!!" as message;
